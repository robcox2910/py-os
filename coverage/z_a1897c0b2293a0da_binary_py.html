1r"""Binary loader — parse and execute PyBin programs. 

2 

3A real OS stores programs as **binary files** — sequences of bytes that 

4the CPU can decode and execute. Formats like ELF (Linux) and PE (Windows) 

5define headers, sections, and instruction encodings so the OS knows how 

6to load a program into memory and run it. 

7 

8Our PyBin format is a simplified version of this idea: 

9 

10- A **magic number** identifies the file as a valid PyBin program 

11 (just like ``\x7fELF`` marks a Linux binary). 

12- A **header** tells the loader the program's name and how many 

13 instructions it contains. 

14- A sequence of **instructions** (bytecodes) that the loader can 

15 interpret and execute. 

16 

17Think of it like recipe cards vs knowing recipes by heart: 

18 

19- A **built-in program** (like ``hello``) is a recipe you've memorised. 

20- A **binary file** is a recipe written on a card — you have to read 

21 the card, understand the steps, and then follow them. 

22 

23The loader is the cook: it reads the card (parses the binary), verifies 

24it looks right (checks the magic number), and follows the steps 

25(executes the instructions). 

26"""

140 def build(self) -> bytes: 

141 """Serialize the program into PyBin binary format. 

142 

143 Layout (little-endian): 

144 - 4 bytes: magic (b"PYBN") 

145 - 2 bytes: version 

146 - 2 bytes: name length, followed by name as UTF-8 

147 - 4 bytes: number of instructions 

148 - Per instruction: 

149 - 1 byte: opcode 

150 - 1 byte: number of arguments 

151 - Per argument: 

152 - 1 byte: type tag (0=int, 1=string) 

153 - If int: 4 bytes (little-endian signed) 

154 - If string: 2 bytes length + UTF-8 bytes 

155 """ 

156 buf = bytearray() 

157 # Header 

158 buf.extend(PYBIN_MAGIC) 

159 buf.extend(struct.pack("<H", PYBIN_VERSION)) 

160 name_bytes = self._name.encode("utf-8") 

161 buf.extend(struct.pack("<H", len(name_bytes))) 

162 buf.extend(name_bytes) 

163 buf.extend(struct.pack("<I", len(self._instructions))) 

164 # Instructions 

165 for inst in self._instructions: 

166 buf.append(inst.opcode) 

167 buf.append(len(inst.args)) 

168 for arg in inst.args: 

169 if isinstance(arg, int): 

170 buf.append(_ARG_TYPE_INT) 

171 buf.extend(struct.pack("<i", arg)) 

172 else: 

173 buf.append(_ARG_TYPE_STRING) 

174 arg_bytes = str(arg).encode("utf-8") 

175 buf.extend(struct.pack("<H", len(arg_bytes))) 

176 buf.extend(arg_bytes) 

177 return bytes(buf)

183 @staticmethod 

184 def parse(data: bytes) -> tuple[PyBinHeader, list[Instruction]]: 

185 """Parse a PyBin binary into a header and instruction list. 

186 

187 Args: 

188 data: The raw binary data. 

189 

190 Returns: 

191 A tuple of (header, instructions). 

192 

193 Raises: 

194 BinaryLoaderError: If the data is invalid. 

195 

196 """ 

197 if len(data) < len(PYBIN_MAGIC): 

198 msg = "Data too short to be a valid PyBin binary" 

199 raise BinaryLoaderError(msg) 

200 magic = data[: len(PYBIN_MAGIC)] 

201 if magic != PYBIN_MAGIC: 

202 msg = f"Invalid magic: expected {PYBIN_MAGIC!r}, got {magic!r}" 

203 raise BinaryLoaderError(msg) 

204 offset = len(PYBIN_MAGIC) 

205 try: 

206 header, offset = _parse_header(data, offset) 

207 instructions = _parse_instructions(data, offset, header.num_instructions) 

208 except struct.error as e: 

209 msg = f"Truncated or corrupt binary: {e}" 

210 raise BinaryLoaderError(msg) from e 

211 return header, instructions

237def _parse_header(data: bytes, offset: int) -> tuple[PyBinHeader, int]: 

238 """Parse the PyBin header after the magic bytes.""" 

239 (version,) = struct.unpack_from("<H", data, offset) 

240 offset += 2 

241 (name_len,) = struct.unpack_from("<H", data, offset) 

242 offset += 2 

243 name = data[offset : offset + name_len].decode("utf-8") 

244 offset += name_len 

245 (num_instructions,) = struct.unpack_from("<I", data, offset) 

246 offset += 4 

247 header = PyBinHeader( 

248 magic=PYBIN_MAGIC, 

249 version=version, 

250 name=name, 

251 num_instructions=num_instructions, 

252 ) 

253 return header, offset

272def _parse_instructions(data: bytes, offset: int, count: int) -> list[Instruction]: 

273 """Parse a sequence of instructions from the binary.""" 

274 instructions: list[Instruction] = [] 

275 for _ in range(count): 

276 opcode = Opcode(data[offset]) 

277 num_args = data[offset + 1] 

278 offset += 2 

279 args: list[str | int] = [] 

280 for _ in range(num_args): 

281 arg, offset = _parse_arg(data, offset) 

282 args.append(arg) 

283 instructions.append(Instruction(opcode=opcode, args=tuple(args)))
