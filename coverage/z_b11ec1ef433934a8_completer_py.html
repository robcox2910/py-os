1"""Context-aware tab completer for the PyOS shell. 

2 

3The completer separates **what to complete** (pure logic, fully 

4testable) from **how to wire it** (readline integration in the REPL). 

5 

6The ``complete(text, state)`` method is the readline callback. It 

7delegates to ``completions(text, line)`` which analyses the input 

8context and returns a list of candidate strings. 

9"""

63class Completer: 

64 """Context-aware tab completer for the PyOS shell.""" 

65 

66 def __init__(self, shell: Shell) -> None: 

67 """Create a completer attached to a shell instance. 

68 

69 Args: 

70 shell: The shell whose commands, kernel, and programs 

71 are used to generate completion candidates. 

72 

73 """ 

74 self._shell = shell 

75 self._kernel: Kernel = shell.kernel 

76 

77 def complete(self, text: str, state: int) -> str | None: 

78 """Readline callback — return the *state*-th candidate for *text*. 

79 

80 Args: 

81 text: The partial word being completed. 

82 state: Index into the candidate list (0, 1, 2, …). 

83 

84 Returns: 

85 The candidate at *state*, or ``None`` when exhausted. 

86 

87 """ 

88 line = readline.get_line_buffer() 

89 candidates = self.completions(text, line) 

90 if state < len(candidates): 

91 return candidates[state] 

92 return None 

93 

94 def completions(self, text: str, line: str) -> list[str]: 

95 """Return completion candidates based on context. 

96 

97 Args: 

98 text: The partial word under the cursor. 

99 line: The full input line so far. 

100 

101 Returns: 

102 Sorted list of matching candidates. 

103 

104 """ 

105 words = line.lstrip().split() 

106 

107 # No words yet, or still typing the first word → command completion 

108 if not words or (len(words) == 1 and not line.endswith(" ")): 

109 return self._complete_commands(text) 

110 

111 return self._complete_argument(words, text, line) 

112 

113 # -- private completers ------------------------------------------------ 

114 

115 def _complete_argument( # noqa: PLR0911 

116 self, words: list[str], text: str, line: str 

117 ) -> list[str]: 

118 """Dispatch argument completion based on the command and context.""" 

119 cmd = words[0] 

120 

121 # Second word for commands that have subcommands 

122 if cmd in _SUBCOMMANDS and ( 

123 len(words) == 1 or (len(words) == _MIN_WORDS_FOR_SIGNAL and not line.endswith(" ")) 

124 ): 

125 return self._complete_subcommands(cmd, text) 

126 

127 # After 'run' → program names 

128 if cmd == "run": 

129 return self._complete_programs(text) 

130 

131 # After 'unset' → env var names 

132 if cmd == "unset": 

133 return self._complete_env_vars(text) 

134 

135 # After 'signal'/'handle' with pid already given → signal names 

136 if cmd in ("signal", "handle") and len(words) >= _MIN_WORDS_FOR_SIGNAL: 

137 return self._complete_signals(text) 

138 

139 # Dollar-prefix in any position → env var names with $ prefix 

140 if text.startswith("$"): 

141 return self._complete_dollar_vars(text) 

142 

143 # Text starts with / or command takes paths → path completion 

144 if text.startswith("/") or cmd in _PATH_COMMANDS: 

145 return self._complete_paths(text) 

146 

147 return [] 

148 

149 def _complete_commands(self, text: str) -> list[str]: 

150 """Complete command names from the shell's dispatch table.""" 

151 return [cmd for cmd in self._shell.command_names if cmd.startswith(text)] 

152 

153 @staticmethod 

154 def _complete_subcommands(cmd: str, text: str) -> list[str]: 

155 """Complete subcommands for a command that has them.""" 

156 return sorted(sub for sub in _SUBCOMMANDS[cmd] if sub.startswith(text)) 

157 

158 def _complete_paths(self, text: str) -> list[str]: 

159 """Complete filesystem paths. 

160 

161 Split the partial path into a directory and a name prefix, 

162 list the directory via SYS_LIST_DIR, and filter by prefix. 

163 Directories get a trailing ``/`` suffix. 

164 

165 For /proc paths, use SYS_PROC_LIST instead of SYS_LIST_DIR. 

166 """ 

167 if "/" not in text: 167 ↛ 168line 167 didn't jump to line 168 because the condition on line 167 was never true

168 return [] 

169 

170 # Split "/foo/ba" into dir="/foo" prefix="ba" 

171 last_slash = text.rfind("/") 

172 directory = text[: last_slash + 1] or "/" 

173 prefix = text[last_slash + 1 :] 

174 

175 syscall = ( 

176 SyscallNumber.SYS_PROC_LIST 

177 if directory.startswith("/proc") 

178 else SyscallNumber.SYS_LIST_DIR 

179 ) 

180 try: 

181 entries: list[str] = self._kernel.syscall(syscall, path=directory) 

182 except SyscallError: 

183 return [] 

184 

185 candidates: list[str] = [] 

186 for entry in entries: 

187 if entry.startswith(prefix): 

188 full = f"{directory.rstrip('/')}/{entry}" 

189 # Check if this entry is itself a directory 

190 if self._is_directory(full): 

191 full += "/" 

192 candidates.append(full) 

193 

194 return sorted(candidates) 

195 

196 def _is_directory(self, path: str) -> bool: 

197 """Return True if *path* is a directory in the filesystem.""" 

198 syscall = ( 

199 SyscallNumber.SYS_PROC_LIST if path.startswith("/proc") else SyscallNumber.SYS_LIST_DIR 

200 ) 

201 try: 

202 self._kernel.syscall(syscall, path=path) 

203 except (SyscallError, NotADirectoryError): 

204 return False 

205 return True

207 def _complete_programs(self, text: str) -> list[str]: 

208 """Complete built-in program names after 'run'.""" 

209 return [name for name in self._shell.builtin_program_names if name.startswith(text)] 

210 

211 def _complete_env_vars(self, text: str) -> list[str]: 

212 """Complete environment variable names (without $ prefix).""" 

213 try: 

214 items: list[tuple[str, str]] = self._kernel.syscall(SyscallNumber.SYS_LIST_ENV) 

215 except SyscallError: 

216 return [] 

217 return sorted(key for key, _val in items if key.startswith(text)) 

218 

219 def _complete_dollar_vars(self, text: str) -> list[str]: 

220 """Complete $VAR references with the dollar prefix.""" 

221 prefix = text[1:] # strip leading $ 

222 try: 

223 items: list[tuple[str, str]] = self._kernel.syscall(SyscallNumber.SYS_LIST_ENV) 

224 except SyscallError: 

225 return [] 

226 return sorted(f"${key}" for key, _val in items if key.startswith(prefix)) 

227 

228 @staticmethod 

229 def _complete_signals(text: str) -> list[str]: 

230 """Complete signal names from the Signal enum.""" 

231 return sorted(sig.name for sig in Signal if sig.name.startswith(text))
